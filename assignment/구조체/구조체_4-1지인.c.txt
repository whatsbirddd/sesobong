#pragma warning(disable:4996)
#include<stdio.h> 
#include<string.h>

struct car {//오제이는 나쁜놈
	int carnum;//차번호
	int num;//학번
	int year[2], month[2], day[2], hour[2], min[2];//날짜,시간

};
int main() {
	struct car badcar[100];//입력되는 주차의 개수는 최대 100이므로
	int numnum[100], newnum[100];
	int n, tmp, count = 0;
	int i, j = 0, cnt = 0, ncnt = 0;

	scanf("%d", &n);//차 몇개인지 입력받기
	for (i = 0; i < n; i++) {//차 개수만큼 반복
		scanf("%d %d", &badcar[i].carnum, &badcar[i].num);//차번호하고 학번입력
		scanf(" %d-%d-%d %d:%d", &badcar[i].year[0], &badcar[i].month[0], &badcar[i].day[0], &badcar[i].hour[0], &badcar[i].min[0]);//입차시간
		scanf(" %d-%d-%d %d:%d", &badcar[i].year[1], &badcar[i].month[1], &badcar[i].day[1], &badcar[i].hour[1], &badcar[i].min[1]);//출차시간
	}


	for (i = 0; i < n; i++) {//이거 왜이렇게 짰을까?..바보같다...
		j = 0;
		while (badcar[i].num != badcar[j].num) {//배열내에서 중복된수를 제외한 배열 만들기...
			j++;

		}
		if (j == i) {//만약 같다면..
			numnum[cnt] = badcar[i].num;//새로운 배열에넣기.. 
			cnt++;//배열의 크기를 측정하기 위해서
		}

	}

	for (i = 0; i < cnt; i++) {//새로운 배열 크기만큼 반복..
		count = 0;
		for (j = 0; j < n; j++) {
			if (numnum[i] == badcar[j].num) {//만약 위에서 새롭게 정렬한 배열과 같다면
				count++;//갯수세고
			}
		}
		if (count > 1) {
			newnum[ncnt] = numnum[i];//배열에 집어넣기
			ncnt++;//갯수세주기
		}
	}
	for (i = 0; i < ncnt; i++) {
		for (j = i + 1; j < ncnt; j++) {//오름차순으로 정리
			if (newnum[i] > newnum[j]) {//더 크면
				tmp = newnum[i];//swap
				newnum[i] = newnum[j];
				newnum[j] = tmp;
			}
		}

	}
	for (i = 0; i < ncnt; i++) {
		printf("%d\n", newnum[i]);//출력해주기
	}
}

			if (c[i].num > c[j].num) {
				tmp = c[i];

			}

		}
	}
	int current_num = c[0].num;

#pragma warning(disable:4996)
#include<stdio.h> 
#include<string.h>

struct hotel_info {//구조체
	char name[31];//호텔 이름
	int grade;//등급
	double rep;//평판
	double dis;//거리
	char bf;//조식 여부
};

int in_hotel_info(struct hotel_info* p) {
	int cnt = 0;

	scanf("%s", &p->name);//이름 입력받기
	while (strcmp(p->name, "0")) {//0일때까지 반복
		scanf("%d %lf %lf", &p->grade, &p->rep, &p->dis);//나머지 정보 입력받기
		getchar();
		scanf("%c", &p->bf);//조식여부
		p++;//늘려나가기
		cnt++;//호텔 갯수세기
		scanf("%s", &p->name);//호텔이름 입력받기
	}

	return cnt;//갯수 반환
}

void out_hotel_info(struct hotel_info* p, int N, int G, double D) {
	int i, j;
	struct hotel_info temp, * q;//교환을 위한 구조체와 주소연결을 위해...

	for (i = 0; i < N - 1; i++) {//버블정렬
		for (q = p; q < p + N - 1; q++) {
			if (q->rep < (q + 1)->rep||
				(q->rep == (q + 1)->rep&& strcmp(q->name, (q + 1)->name) > 0)) {//평판비교와 사전식 비교
				temp = *q;
				*q = *(q + 1);//교환하기
				*(q + 1) = temp;//정렬하기 위함
			}
		}
	}

	for (q = p; q < p + N; q++) {//출력
		if (q->grade >= G && q->dis <= D) {//조건에 맞으면
			printf("%s %d %.1f %.1f %c\n", q->name, q->grade, q->rep, q->dis, q->bf);//출력하기
		}
	}
}

int main() {
	int N, G;
	double D;
	struct hotel_info h[100];//구조체

	N = in_hotel_info(h);//호텔 갯수 받기
	scanf("%d %lf", &G, &D);//조건 입력
	out_hotel_info(h, N, G, D);//조건 함수

	return 0;
}
